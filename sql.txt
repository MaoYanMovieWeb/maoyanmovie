
/*创建数据库*/
DROP DATABASE IF EXISTS  maoyanmovie ;
CREATE DATABASE maoyanmovie DEFAULT CHARACTER SET utf8;
USE maoyanmovie;

/*创建用户表*/
CREATE TABLE user (
  uid int(15) NOT NULL AUTO_INCREMENT,
  user_name varchar(255) DEFAULT NULL,
  user_password varchar(255) DEFAULT NULL,
  user_tel varchar(255) DEFAULT NULL,
  user_photo varchar(255) DEFAULT NULL,
  PRIMARY KEY (uid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

/*创建电影表*/
CREATE TABLE movie (
  mid int(15) NOT NULL AUTO_INCREMENT,
  movie_name varchar(255) DEFAULT NULL,
  movie_english varchar(255) DEFAULT NULL,
  movie_time varchar(255) DEFAULT NULL,
  movie_type varchar(255) DEFAULT NULL,
  movie_information varchar(2000) DEFAULT NULL,
  movie_region varchar(255) DEFAULT NULL,
  movie_photo varchar(255) DEFAULT NULL,
  movie_score varchar(255) DEFAULT NULL,
  movie_money varchar(255) DEFAULT NULL,
  PRIMARY KEY (mid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

/*创建订单表*/
CREATE TABLE orders (
  oid int(15) NOT NULL AUTO_INCREMENT,
  uid int(15) DEFAULT NULL,
  mid int(15) DEFAULT NULL,
  cid int(15) DEFAULT NULL,
  order_time varchar(255) DEFAULT NULL,
  PRIMARY KEY (oid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

/*创建演员表*/
CREATE TABLE `actor` (
  aid int(15) NOT NULL AUTO_INCREMENT,
  mid int(15) DEFAULT NULL,
  director_name varchar(255) DEFAULT NULL,
  director_photo varchar(255) DEFAULT NULL,
  actor_name1 varchar(255) DEFAULT NULL,
  actor_photo1 varchar(255) DEFAULT NULL,
  actor_name2 varchar(255) DEFAULT NULL,
  actor_photo2 varchar(255) DEFAULT NULL,
  actor_name3 varchar(255) DEFAULT NULL,
  actor_photo3 varchar(255) DEFAULT NULL,
  actor_name4 varchar(255) DEFAULT NULL,
  actor_photo4 varchar(255) DEFAULT NULL,
  PRIMARY KEY (`aid`)
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8;

/*创建影院表*/
CREATE TABLE cinema (
  cid int(15) NOT NULL AUTO_INCREMENT,
  cinema_name varchar(255) DEFAULT NULL,
  cinema_address varchar(255) DEFAULT NULL,
 cinema_tel varchar(255) DEFAULT NULL,
  PRIMARY KEY (cid)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8;
